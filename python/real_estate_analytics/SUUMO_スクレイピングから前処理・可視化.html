def format_bldg_info(item):
    import re
    bldg_info = {}
    # 建物名
    bldg_info['bldg_name'] = item.find_all("div",{'class': 'cassetteitem_content-title'})[0].text
    # 住所取得
    bldg_info['bldg_address'] = item.find_all("li", {'class': 'cassetteitem_detail-col1'})[0].text
    # 最寄り駅とそこからの距離(徒歩ベース)を取得
    sta_walk = item.find_all("li", {'class': 'cassetteitem_detail-col2'})[0].text.strip('\n').split('\n')
    for sta in sta_walk:
        # 分析対象の駅が出たら、そこの駅名と徒歩距離を指定する
        if re.search('新丸子駅|小杉駅|日吉駅|綱島駅|菊名駅', sta):
            break
    bldg_info['station'] = sta.split(' ')[0].split('/')[1]
    walk = sta.split(' ')[-1]
    # 当該駅からの経路がバスのやつは、一律で[30分]かかるものとする
    if re.search('バス', sta):
        bldg_info['min_walk'] = 30
    else:
        bldg_info['min_walk'] = int(re.sub('[徒歩分]','',walk))
    # 築年数と建物の高さ
    age = item.find_all("li", {'class': 'cassetteitem_detail-col3'})[0].text.strip('\n').split('\n')[0]
    age = age.replace('新築','0')
    bldg_info['age'] = int(re.sub('[築年]','',age))
    bldg_info['height'] = item.find_all("li", {'class': 'cassetteitem_detail-col3'})[0].text.strip('\n').split('\n')[-1]
    return bldg_info

def format_room_info(item):
    url_base = 'https://suumo.jp'
    room_info = {}
    # 部屋別にほしい情報を取得するiterを回す
    for ind, room in enumerate(item.find_all('tbody')):
        room_info[ind] = {}
        room_info[ind]['floor'] = room.find_all('td')[2].text.replace('\t','').replace('\n','')
        room_info[ind]['url'] = url_base + room.find('td',{'class':'ui-text--midium ui-text--bold'}).find('a').get('href')
        try:
            room_info[ind]['price'] = room.find('span',{'class':'cassetteitem_price cassetteitem_price--rent'}).text
            room_info[ind]['kanrihi'] = room.find('span',{'class':'cassetteitem_price cassetteitem_price--administration'}).text
            room_info[ind]['shikikin'] = room.find('span',{'class':'cassetteitem_price cassetteitem_price--deposit'}).text
            room_info[ind]['reikin'] = room.find('span',{'class':'cassetteitem_price cassetteitem_price--gratuity'}).text
            room_info[ind]['room_type'] = room.find('span',{'class':'cassetteitem_madori'}).text
            room_info[ind]['floor_m2'] = room.find('span',{'class':'cassetteitem_menseki'}).text
        except:
            room_desc_l = [x.text for x in room.find_all('td')][2:8]
            room_info[ind]['price'] = room_desc_l[1]
            room_info[ind]['kanrihi'] = room_desc_l[2]
            initial_costs = room_desc_l[3]
            room_info[ind]['shikikin'] = initial_costs.split('/')[0]
            room_info[ind]['reikin'] = initial_costs.split('/')[1]
            room_info[ind]['room_type'] = room_desc_l[4].replace('\t','').replace('\n','')
            room_info[ind]['floor_m2'] = room_desc_l[5].replace('\t','').replace('\n','')
            room_info[ind]['url'] = url_base + room.find('td',{'class':'ui-text--midium ui-text--bold'}).find('a').get('href')
    return room_info

for i, file in tqdm(enumerate(files)):
    with open(file,'r') as f:
        soup = BeautifulSoup(f.read(), 'lxml')
    df = pd.DataFrame()
    summary = soup.find("div", {'id': 'js-bukkenList'})
    cassetteitems = summary.find_all("div", {'class': 'cassetteitem'})
    for item in cassetteitems:
        # 建物関係の情報を取得する
        bldg_info = format_bldg_info(item)
        # 部屋個別の情報を取得する
        room_info = format_room_info(item)
        # 部屋個別の情報から、pandas dfを作成
        bldg_df = pd.DataFrame.from_dict(room_info, orient='index')
        # dfに建物の情報を付与する
        for key in bldg_info.keys():
            bldg_df.loc[:,key] = bldg_info[key]
        df = pd.concat([df, bldg_df]).reset_index(drop=True)
    df.to_csv('formatted_data/pn-{}.csv'.format(i+1), index=False)

import re
import doctest

def format_price(price):
    """
    >>> format_price('10万円')
    100000.0
    >>> format_price('5000円')
    5000.0
    """
    if price == '-':
        return 0
    elif re.search('万円',price):
        return float(re.sub('万円','', price)) * 10000
    else:
        return float(re.sub('円','', price))

    
def format_common_data(data):
    """
    >>> format_common_data('45.2m2')
    45.2
    >>> format_common_data('5階建')
    5.0
    """
    if re.search('-',data):
        # 特殊措置
        return int(data[0])
    else:
        ret = re.sub('地下\d地上','',data)
        ret = re.sub('[階建]','',ret)
        ret = ret.replace('m2','')
        ret = float(ret)
        return ret
