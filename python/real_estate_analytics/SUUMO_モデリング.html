param_df.loc[:,'mean_score_diff'] = param_df.mean_train_score - param_df.mean_test_score

def plot_param_vs_score(param_col):
    plt.plot(param_df[param_col], param_df.mean_train_score, '.', color='blue', label='train_data_score')
    plt.plot(param_df[param_col], param_df.mean_test_score, '.', color='red', label='test_data_score')
    plt.title(param_col)
    plt.legend(loc='center left', bbox_to_anchor=(1, 0.5))
    plt.show()
    plt.plot(param_df[param_col], param_df.mean_score_diff, '.', color='black', label='score_diff')
    plt.title(param_col)
    plt.legend(loc='center left', bbox_to_anchor=(1, 0.5))
    plt.show()
    
    
cols = ['param_n_estimators', 'param_min_samples_split', 'param_min_samples_leaf','param_max_depth']

for param_col in cols:
    plot_param_vs_score(param_col)
